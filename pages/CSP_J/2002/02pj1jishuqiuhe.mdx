#### 题解代码：
```cpp showLineNumbers {10,12,13}
#include <bits/stdc++.h>
using namespace std;

int k, n = 0;
double s = 0;	

int main()
{
	scanf("%d", &k);
	while(s <= k)
	{
		n++;
		s += 1/((double)n);
	} 

	printf("%d\n", n);	
	return 0;	
}
```
