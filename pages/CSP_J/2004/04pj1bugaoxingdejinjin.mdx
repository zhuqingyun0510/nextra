#### 题解代码：
```cpp showLineNumbers
#include <bits/stdc++.h>
using namespace std;

int a, b, c, ma = -1, day;
bool flag = false;  // 是否找到

int main()
{
    for(int i = 1; i <= 7; i++)  // i从1开始，表示星期一
    {
        scanf("%d %d", &a, &b);
        c = a + b;
        if( (c > 8) && (c > ma) )  // c>ma表示只有比ma大，才会更新日期。即使相等也不更新
        {
            ma = c;
            day = i;  // 记录最不开心的那一天
            flag = true;
        }
    }

    if(!flag)  printf("0\n");
    else  printf("%d", day);

    return 0;
}
```
